        f.setSize(300,300);
        f.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
        f.setLayout(new BorderLayout());
        JButton b=new JButton("Btn");
        JLabel c=new JLabel("√Årea central",SwingConstants.CENTER);
        f.add(b,BorderLayout.NORTH);

    }
}
